  3 /**
  4  * Creates a generator.
  5  * @param {number} start The first number in the generated sequence.
  6  * @param {number} end  The last number in the generated sequence.
  7  * @param {number} step The interval between numbers in the generated sequence.
  8  */
  9 function* sequence(start = 0, end = Infinity, step = 1) {
 10   for (let current = start; current < end; current = current + step) {
 11     yield current;
 12   }
 13 }
 14 
 15 /**
 16  * Creates a generator using an iterator over the numbers 1..5.
 17  */
 18 function* genUsingIterator() {
 19   yield* [1, 2, 3, 4, 5];
 20 }

 27 /**
 28  * Prints the values generated by the specified generator.
 29  *
 30  * @param {Generator} theGen Generates the values that are printed.
 31  */
 32 function printGeneratedVals(theGen) {
 33   let current = theGen.next();
 34   while (!current.done) {
 35     console.log(current.value);
 36     current = theGen.next();
 37   }
 38 }
